Convert data to wide format

sfp_care_wf <-
  sfp_care %>%
  as.data.frame() %>%
  reshape(idvar = "ID", timevar = "Episode", direction = "wide") %>%
  group_by(ID) %>% 
  mutate(
    sens_M = mean(c(sens_M.P, sens_M.R1, sens_M.R2), na.rm = TRUE),
    intrus_M = mean(c(intrus_M.P, intrus_M.R1, intrus_M.R2), na.rm = TRUE),
